37/** @brief capture_shell_tioctl() - capture shell terminal settings

76/** @brief set_sane_tioctl() - set terminal to sane settings for C-MENU
77 @param t_p - pointer to termios structure to modify
78 @return - true on success
79 @note - sets terminal to sane settings for C-MENU applications */
80bool set_sane_tioctl(struct termios *t_p) {
81 tcgetattr(0, t_p);
82 t_p->c_iflag &= ~(IGNBRK | BRKINT | PARMRK | INPCK | ISTRIP | INLCR |
83 IGNCR | ICRNL | IXON | IXOFF);
84 t_p->c_iflag |= IUTF8;
85 t_p->c_oflag |= OPOST | OCRNL;
86 t_p->c_lflag &= ~(ECHO | ECHONL | ICANON | IEXTEN);
87 t_p->c_lflag |= (ISIG | ICANON | IEXTEN | ECHO | ECHOE | ECHOK);
88 t_p->c_cflag &= ~(CSIZE | PARENB);
89 t_p->c_cflag |= CS8 | CLOCAL | CREAD;
90 tcsetattr(0, TCSANOW, t_p);
91 return true;
92}
93/** @brief mk_raw_tioctl() - set terminal to raw mode
94 @param t_p - pointer to termios structure to modify
95 @return - true on success
96 @note - unlike cfmakeraw(), this leaves ISIG enabled.
97 @note - cfmakeraw() disables ISIG, which prevents C-MENU from handling
98 signals like Ctrl-C. Instead of using cfmakeraw(), we manually set the
99 terminal to raw mode while leaving ISIG enabled.
100 @note - the following code is equivalent to cfmakeraw() but with ISIG left
101 enabled:
102 @code
103 t_p->c_iflag &= ~(IGNBRK | BRKINT | PARMRK | INPCK | ISTRIP | INLCR |
104 IGNCR | ICRNL | IXON | IXOFF);
105 t_p->c_lflag &= ~(ECHO | ECHONL | ICANON | IEXTEN);
106 t_p->c_cflag &= ~(CSIZE | PARENB);
107 t_p->c_cflag |= CS8 | CLOCAL;
108 @endcode
109 */
110bool mk_raw_tioctl(struct termios *t_p) {
111 tcgetattr(0, t_p);
112 t_p->c_lflag |= ISIG;
113 t_p->c_lflag &= ~(ECHO | ICANON);
114 t_p->c_cc[VMIN] = 1;
115 t_p->c_cc[VTIME] = 0;
116 tcsetattr(0, TCSAFLUSH, t_p);
117 return true;
118}
119/** @brief sget single character from terminal in raw mode
120 @return - the character read from terminal
121 @note restores terminal settings */
122char di_getch() {
123 struct termios org_tioctl, new_tioctl;
124 char buf;
125
126 if (tcgetattr(2, &org_tioctl) == -1) {
127 fprintf(stderr, "\ndi_getch: tcgetattr failed\n");
128 return (0);
129 }
130 new_tioctl = org_tioctl;
131 new_tioctl.c_lflag &= ~(ECHO | ICANON);
132 new_tioctl.c_cc[VMIN] = 1;
133 new_tioctl.c_cc[VTIME] = 0;
134 tcsetattr(2, TCSAFLUSH, &new_tioctl);
135 read(2, &buf, 1);
136 tcsetattr(2, TCSAFLUSH, &org_tioctl);
137 return (buf);
138}
