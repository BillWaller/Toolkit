526 char *token;
527 char *s;
528 int cols = 0;
529 int i, l;
530 int in_line_num = 0;
531 char in_buf[BUFSIZ];
532 char tmp_buf[BUFSIZ];
533 char *tmp_buf_p;
534 char tmp_str[BUFSIZ];
535 char delim[5];
536 char directive;
537
538 form_desc_fp = fopen(form->mapp_spec, "r");
539 if (form_desc_fp == nullptr) {
540 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__, __LINE__ - 2);
541 strnz__cpy(em1, "fopen ", MAXLEN - 1);
542 strnz__cat(em1, form->mapp_spec, MAXLEN - 1);
543 strerror_r(errno, em2, MAXLEN);

546 }
547 for (i = 0; i < MAXFIELDS; i++) {
548 form->field[i] = (Field *)calloc(1, sizeof(Field));
549 if (!form->field[i]) {

551 abend(EXIT_FAILURE, tmp_str);
552 }
553 }
554 for (i = 0; i < MAXFIELDS; i++) {
555 form->text[i] = (Text *)calloc(1, sizeof(Text));
556 if (!form->field[i]) {
557 sprintf(tmp_str, "FORM: calloc failed for text");
558 abend(EXIT_FAILURE, tmp_str);
559 }
560 }
561 form->didx = 0;
562 form->fidx = 0;
563 form->fcnt = 0;
564 form->cols = 34;
565 while ((fgets(in_buf, MAXLEN, form_desc_fp)) != nullptr) {
566 s = in_buf;
567 in_line_num++;
568 l = trim(in_buf);
569 if (l == 0)
570 continue;
571 if (*s == D_COMMENT)
572 continue;
573 delim[0] = '\n';
574 delim[1] = in_buf[1];
575 delim[2] = '\0';
576 strnz__cpy(tmp_buf, in_buf, MAXLEN - 1);
577 tmp_buf_p = tmp_buf;
578 if (!(token = strtok(tmp_buf_p, delim))) {
579 continue;
580 }
581 directive = *token;
582 switch ((int)directive) {
583 case D_COMMENT:
584 break;
585 case D_CALC:
586 form->f_calculate = true;
587 break;
588 case D_QUERY:
589 form->f_query = true;
590 break;
591 case D_CMD:
592 if (!(token = strtok(nullptr, delim))) {
593 form_desc_error(in_line_num, in_buf,
594 "FORM: receiver_cmd delimiter");
595 continue;
596 }
597 strnz__cpy(form->receiver_cmd, token, MAXLEN - 1);
598 break;
599 case D_HELP:
600 if (!(token = strtok(nullptr, delim))) {
601 form_desc_error(in_line_num, in_buf,
602 "FORM: help_spec delimiter");
603 }
604 strnz__cpy(form->help_spec, token, MAXLEN - 1);
605 break;
606 case D_FIELD:
607 if (form->field[form->fidx] == nullptr) {
608 sprintf(tmp_str, "FORM: calloc failed for fields");
609 abend(EXIT_FAILURE, tmp_str);
610 }
611 if (!(token = strtok(nullptr, delim))) {
612 form_desc_error(in_line_num, in_buf,
613 "FORM: line number delimiter");
614 return 1;
615 }
616 form->field[form->fidx]->line = atoi(token);
617 if (form->field[form->fidx]->line < 0 ||
618 form->field[form->fidx]->line >= MAXFIELDS) {
619 form_desc_error(in_line_num, in_buf,
620 "FORM: invalid line number");
621 return 1;
622 }
623 if (!(token = strtok(nullptr, delim))) {
624 form_desc_error(in_line_num, in_buf,
625 "FORM: column number delimiter");
626 return 1;
627 }
628 form->field[form->fidx]->col = atoi(token);
629 if (form->field[form->fidx]->col < 0 ||
630 form->field[form->fidx]->col >= TBL_COLS) {
631 form_desc_error(in_line_num, in_buf,

633 break;
634 }

636 strnz__cpy(tmp_str, in_buf, MAXLEN - 1);

638 break;
639 }
640 form->field[form->fidx]->len = atoi(token);
641 if (form->field[form->fidx]->len < 0 ||
642 form->field[form->fidx]->len > TBL_COLS) {
643 form_desc_error(in_line_num, in_buf, "FORM: invalid length");
644 break;
645 }

647 form_desc_error(in_line_num, in_buf,
648 "FORM: validation code delimiter");
649 break;
650 }
651 form->field[form->fidx]->ff = -1;
652 for (i = 0; i < FF_INVALID; i++) {

656 form->field[form->fidx]->ff = i;

658 }
659 }
660 if (form->field[form->fidx]->ff < 0 ||
661 form->field[form->fidx]->ff >= FF_INVALID) {

663 "FORM: invalid format code");

665 }
666 cols =
667 form->field[form->fidx]->col + form->field[form->fidx]->len + 1;
668 if (cols > form->cols)

672 break;
673 case D_TEXT:
674 if (form->text[form->didx] == nullptr) {
675 sprintf(tmp_str, "FORM: calloc failed for text");
676 abend(EXIT_FAILURE, tmp_str);
677 }
678 if (!(token = strtok(nullptr, delim))) {
679 form_desc_error(in_line_num, in_buf,
680 "FORM: line number delimiter");
681 break;
682 }
683 form->text[form->didx]->line = atoi(token);
684 if (form->text[form->didx]->line < 0 ||
685 form->text[form->didx]->line >= MAXFIELDS) {
686 form_desc_error(in_line_num, in_buf,
687 "FORM: invalid line number");
688 break;
689 }
690 if (!(token = strtok(nullptr, delim))) {
691 form_desc_error(in_line_num, in_buf,
692 "FORM: column number delimiter");
693 break;
694 }
695 form->text[form->didx]->col = atoi(token);
696 if (form->text[form->didx]->col < 0 ||
697 form->text[form->didx]->col >= TBL_COLS) {
698 form_desc_error(in_line_num, in_buf,
699 "FORM: invalid column number");
700 break;
701 }
702 if (!(token = strtok(nullptr, delim))) {
703 form_desc_error(in_line_num, in_buf, "FORM: text delimiter");
704 break;
705 }
706 strnz__cpy(form->text[form->didx]->str, token, MAXLEN - 1);
707 form->text[form->didx]->len = strlen(form->text[form->didx]->str);
708 if (form->text[form->didx]->len < 0 ||
709 form->text[form->didx]->len > TBL_COLS) {
710 form_desc_error(in_line_num, in_buf, "FORM: invalid length");
711 break;
712 }
713 cols =
714 form->text[form->didx]->col + form->text[form->didx]->len + 1;
715 if (cols > form->cols)
716 form->cols = cols;
717 form->didx++;
718 form->dcnt = form->didx;
719 break;
720 case D_HEADER:
721 if ((token = strtok(nullptr, delim))) {

723 }
724 break;
725 default:
726 form_desc_error(in_line_num, in_buf, "invalid directive");
727 break;
728 }
729 }
730 fclose(form_desc_fp);
731 if (form->didx < 1 && form->fidx < 1) {
