22/** @brief Initialize C-Menu View in full screen mode.
23 @note This function sets up the view structure for full screen mode and
24 creates a new pad for the view.
25 @param init Pointer to the Init structure containing view settings.
26 @return 0 on success, -1 on failure.
27 */
28int init_view_full_screen(Init *init) {
29 view = init->view;
30 view->f_full_screen = true;
31 getmaxyx(stdscr, view->lines, view->cols);
32 view->pminrow = 0;
33 view->pmincol = 0;
34 view->sminrow = 0;
35 view->smincol = 0;
36 view->scroll_lines = view->lines - 1;
37 view->cmd_line = view->lines - 1;
38 view->smaxrow = view->lines - 1;
39 view->smaxcol = view->cols - 1;
40 view->win = newpad(view->lines, PAD_COLS);
41 if (view->win == nullptr) {
42 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__, __LINE__ - 2);
43 ssnprintf(em1, MAXLEN - 1, "newpad(%d, %d) failed", view->lines,
44 PAD_COLS);
45 em2[0] = '\0';
46 display_error(em0, em1, em2, nullptr);
47 abend(-1, "init_view_full_screen: newpad() failed");
48 }
49 view->box = nullptr;
50 wbkgrnd(view->box, &CCC_BOX);
51 if (view->tab_stop <= 0)
52 view->tab_stop = TABSIZE;
53 set_tabsize(view->tab_stop);
54 wsetscrreg(view->win, 0, view->scroll_lines - 1);
55 scrollok(view->win, true);
56 keypad(view->win, true);
57 idlok(view->win, false);
58 idcok(view->win, false);
59 return 0;
60}

130/** @brief Initialize the input for a C-Menu View.
131 @details This function initializes the input for view, which can be a file,
132 standard input, or a provider command to be initiated by view. It handles
133 different input sources and sets up the necessary file descriptors and memory
134 mapping for efficient access.
135 @param view Pointer to the View structure to be initialized.
136 @param file_name Name of the input file or "-" for standard input.
137 @return true on success, false on failure.
138 @note if a provider command is specified, set up a pipe to read its output.
139 A child process is spawned, and view, the parent process, reads from the
140 pipe.
141 @note If input is from a pipe or standard input, clone it to a temporary
142 file. This allows for memory-mapping the input later. It does not support
143 real-time updates to the input, but it allows for efficient access to the
144 data.
145 */
146bool view_init_input(View *view, char *file_name) {
147 struct stat sb;
148 int idx = 0;
149 pid_t pid;
150 int pipe_fd[2];
151 char *s_argv[MAXARGS];
152 char tmp_str[MAXLEN];
153 view->f_in_pipe = false;
154 if (strcmp(file_name, "-") == 0) {
155 file_name = "/dev/stdin";
156 view->f_in_pipe = true;
157 }
158 if (view->provider_cmd[0] != '\0') {
159 str_to_args(s_argv, view->provider_cmd, MAXARGS - 1);
160 if (pipe(pipe_fd) == -1) {
161 Perror("pipe(pipe_fd) failed in init_view");
162 return (1);
163 }
164 if ((pid = fork()) == -1) {
165 Perror("fork() failed in init_view");
166 return (1);
167 }
168 if (pid == 0) { // Child
169 close(pipe_fd[P_READ]);
170 dup2(pipe_fd[P_WRITE], STDOUT_FILENO);
171 close(pipe_fd[P_WRITE]);
172 execvp(s_argv[0], s_argv);
173 strnz__cpy(tmp_str, "Can't exec view start cmd: ", MAXLEN - 1);
174 strnz__cat(tmp_str, s_argv[0], MAXLEN - 1);
175 Perror(tmp_str);
176 exit(EXIT_FAILURE);
177 }
178 // Back to parent
179 close(pipe_fd[P_WRITE]);
180 dup2(pipe_fd[P_READ], STDIN_FILENO);
181 view->in_fd = dup(STDIN_FILENO);
182 view->f_in_pipe = true;
183 } else {
184 if (view->f_in_pipe)
185 view->in_fd = dup(STDIN_FILENO);
186 else {
187 /** Open the input file for reading and get its size. */
188 expand_tilde(file_name, MAXLEN - 1);
189 view->in_fd = open(file_name, O_RDONLY);
190 if (view->in_fd == -1) {
191 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__,
192 __LINE__ - 2);
193 ssnprintf(em1, MAXLEN - 1, "open %s", file_name);
194 strerror_r(errno, em2, MAXLEN);
195 display_error(em0, em1, em2, nullptr);
196 return false;
197 }
198 if (fstat(view->in_fd, &sb) == -1) {
199 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__,
200 __LINE__ - 1);
201 ssnprintf(em1, MAXLEN - 1, "fstat %s", file_name);
202 strerror_r(errno, em2, MAXLEN);
203 display_error(em0, em1, em2, nullptr);
204 close(view->in_fd);
205 return (EXIT_FAILURE);
206 }
207 view->file_size = sb.st_size;
208 if (view->file_size == 0) {
209 close(view->in_fd);
210 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__,
211 __LINE__ - 1);
212 ssnprintf(em1, MAXLEN - 1, "file %s is empty", file_name);
213 strerror_r(errno, em2, MAXLEN);
214 display_error(em0, em1, em2, nullptr);
215 return (EXIT_FAILURE);
216 }
217 if (!S_ISREG(sb.st_mode))
218 view->f_in_pipe = true;
219 }
220 }
221 if (view->f_in_pipe) {
222 char tmp_filename[] = "/tmp/view_XXXXXX";
223 char buf[VBUFSIZ];
224 ssize_t bytes_read = 0;
225 ssize_t bytes_written = 0;
226 close(view->in_fd);
227 view->in_fd = mkstemp(tmp_filename);
228 if (view->in_fd == -1) {
229 abend(-1, "failed to mkstemp");
230 exit(EXIT_FAILURE);
231 }
232 unlink(tmp_filename);
233 while ((bytes_read = read(STDIN_FILENO, buf, sizeof(buf))) > 0) {
234 if (write(view->in_fd, buf, bytes_read) != bytes_read) {
235 abend(-1, "unable to write tmp");
236 exit(EXIT_FAILURE);
237 }
238 bytes_written += bytes_read;
239 }
240 if (bytes_written == 0) {
241 abend(-1, "unable to read stdin");
242 exit(EXIT_FAILURE);
243 }
244 if (fstat(view->in_fd, &sb) == -1) {
245 abend(-1, "fstat failed");
246 exit(EXIT_FAILURE);
247 }
248 view->file_size = sb.st_size;
249 if (view->file_size == 0) {
250 close(view->in_fd);
251 strnz__cpy(tmp_str, "no standard input", MAXLEN - 1);
252 abend(-1, tmp_str);
253 exit(EXIT_FAILURE);
254 }
255 waitpid(-1, nullptr, 0);
256 }
257 // Memory-map the input file for efficient access.
258 view->buf =
259 mmap(nullptr, view->file_size, PROT_READ, MAP_PRIVATE, view->in_fd, 0);
260 if (view->buf == MAP_FAILED) {
261 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__, __LINE__ - 2);
262 ssnprintf(em1, MAXLEN - 1, "mmap %s", file_name);
263 strerror_r(errno, em2, MAXLEN);
264 display_error(em0, em1, em2, nullptr);
265 close(view->in_fd);
266 return (EXIT_FAILURE);
267 }
268 close(view->in_fd);
269 view->file_size = sb.st_size;
270 view->prev_file_pos = NULL_POSITION;
271 view->buf_curr_ptr = view->buf;
272 if (view->cmd_all[0] != '\0')
273 strnz__cpy(view->cmd, view->cmd_all, MAXLEN - 1);
274 for (idx = 0; idx < NMARKS; idx++)
275 view->mark_tbl[idx] = NULL_POSITION;
276 strnz__cpy(view->cur_file_str, file_name, MAXLEN - 1);
277 base_name(view->file_name, view->cur_file_str);
278 return true;
279}
