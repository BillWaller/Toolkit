453int rgb_to_xterm256_idx(RGB *rgb) {
454 if (rgb->r == rgb->g && rgb->g == rgb->b) {
455 if (rgb->r < 8)
456 return 16;
457 if (rgb->r > 248)
458 return 231;
459 return ((rgb->r - 8) / 10) + 231;
460 } else {
461 int r_index = (rgb->r < 45) ? 0 : (rgb->r - 60) / 40 + 1;
462 int g_index = (rgb->g < 45) ? 0 : (rgb->g - 60) / 40 + 1;
463 int b_index = (rgb->b < 45) ? 0 : (rgb->b - 60) / 40 + 1;
464 return 16 + (36 * r_index) + (6 * g_index) + b_index;
465 }
466}

474RGB xterm256_idx_to_rgb(int idx) {
478 RGB rgb;
479 if (idx > 255)
480 idx = 255;
481 if (idx < 0)
482 idx = 0;
483 rgb.r = rgb.g = rgb.b = 0;
484 if (idx < 16) {
485 rgb.r = StdColors[idx].r;
486 rgb.g = StdColors[idx].g;
487 rgb.b = StdColors[idx].b;
488 } else if (idx >= 16 && idx <= 231) {
489 idx -= 16;
490 rgb.r = (idx / 36) % 6 * 51;
491 rgb.g = (idx / 6) % 6 * 51;
492 rgb.b = (idx % 6) * 51;
493 } else if (idx >= 232 && idx <= 255) {
494 int gray = (idx - 232) * 11;
495 rgb.r = rgb.g = rgb.b = gray;
496 }
497 return rgb;
498}

507void apply_gamma(RGB *rgb) {
508 if (rgb->r == rgb->g && rgb->r == rgb->b) {
509 if (GRAY_GAMMA > 0.0f && GRAY_GAMMA != 1.0f) {
510 rgb->r = (int)(pow((rgb->r / 255.0f), 1.0f / GRAY_GAMMA) * 255.0f);
511 rgb->g = rgb->r;
512 rgb->b = rgb->r;
513 }
514 return;
515 }
516 if (rgb->r != 0 && RED_GAMMA > 0.0f && RED_GAMMA != 1.0f)
517 rgb->r = (int)(pow((rgb->r / 255.0f), 1.0f / RED_GAMMA) * 255.0f);
518 if (rgb->g != 0 && GREEN_GAMMA > 0.0f && GREEN_GAMMA != 1.0f)
519 rgb->g = (int)(pow((rgb->g / 255.0f), 1.0f / GREEN_GAMMA) * 255.0f);
520 if (rgb->b != 0 && BLUE_GAMMA > 0.0f && BLUE_GAMMA != 1.0f)
521 rgb->b = (int)(pow((rgb->b / 255.0f), 1.0f / BLUE_GAMMA) * 255.0f);
522}
