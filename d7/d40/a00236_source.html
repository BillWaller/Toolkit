38/** @brief Main function for whence utility
39 @param argc Argument count
40 @param argv Argument vector
41 @return Exit status
42 options include:
43 -a: Show all matches (default is to show only the first match)
44 -h: Display help message
45 -v: Verbose mode (implies -a)
46 @note This function processes command-line arguments, retrieves the PATH
47 environment variable, and calls the whence function for each specified file.
48 */
49int main(int argc, char **argv) {
50
51 int opt;
52 if (argc < 2)
53 whence_usage();
54 while ((opt = getopt(argc, argv, "ahv")) != -1) {
55 switch (opt) {
56 case 'a':
57 f_all = TRUE;
58 break;
59 case 'h':
60 f_display_help = TRUE;
61 break;
62 case 'v':
63 f_verbose = TRUE;
64 break;
65 default:
66 ABEND(argv[0], FAIL, "unrecognized option");
67 }
68 }
69 if (f_display_help) {
70 whence_usage();
71 normalend();
72 }
73 path_p = getenv("PATH");
74 if (path_p == nullptr)
75 ABEND(argv[0], FAIL, "PATH environment variable not set");
76 if (f_verbose)
77 printf("%s\n", path_p);
78 while (optind < argc) {
79 whence(argv[optind++]);
80 }
81 normalend();
82}

133/** @brief Extract the next directory path from the PATH string
134 @param dp A buffer to store the extracted directory path
135 @param sp A pointer to the current position in the PATH string
136 @return The length of the extracted directory path
137 @note This function takes a buffer and a pointer to the current position in
138 the PATH string, and extracts the next directory path. If the next character
139 in the PATH string is a colon, it treats it as an empty path and uses the
140 current working directory. Otherwise, it copies characters until it reaches a
141 colon or the end of the string, and returns the length of the extracted path.
142 */
143int next_path(char *dp, char **sp) {
144 int dl;
145
146 if (**sp == ':') {
147 (*sp)++;
148 getcwd(dp, PATH_MAX);
149 return (strlen(dp));
150 } else {
151 dl = 0;
152 while (**sp != '\0' && **sp != '\n' && **sp != '\r' && **sp != ':') {
153 *dp++ = *(*sp)++;
154 dl++;
155 }
156 *dp = '\0';
157 if (**sp == ':' && *++(*sp) == '\0')
158 (*sp)--;
159 return (dl);
160 }
161}
162/** @brief Split a file specification into directory and file name components
163 @param file_spec The full file specification to split
164 @param file_path A buffer to store the extracted directory path
165 @param file_name A buffer to store the extracted file name
166 @return 0 on success
167 @note This function takes a file specification, checks if it is a directory,

169 empty, it defaults to the current directory. Otherwise, it splits the file

176 struct stat stat_struct;
177
178 if (stat(file_spec, &stat_struct) != -1)
179 if ((stat_struct.st_mode & S_IFMT) == S_IFDIR) {
180 if (file_spec[strlen(file_path)] != '/')
181 strnz__cat(file_spec, "/", MAXLEN - 1);
182 strnz__cpy(file_path, file_spec, MAXLEN - 1);
183 file_name[0] = '\0';
184 return (0);
185 }
186 if (strlen(file_spec) == 0) {
187 strnz__cpy(file_spec, "./", MAXLEN - 1);
188 file_name[0] = '\0';
189 return (0);
190 }
191 strnz__cpy(tmp_file_spec, file_spec, MAXLEN - 1);
192 last_slash = -1;
193 file_spec_l = strlen(tmp_file_spec);
194 if (file_spec_l > 0) {
195 i = 0;
196 while (i < file_spec_l && tmp_file_spec[i] != '\0') {
197 if (tmp_file_spec[i] == '/') {
198 last_slash = i;
199 break;
200 }
201 i++;
202 }
203 }
204 if (last_slash < 0) {
205 strnz__cpy(file_path, "./", MAXLEN - 1);
206 if (strcmp(file_spec, ".") == 0)
207 file_name[0] = '\0';
208 else
209 strnz__cpy(file_name, tmp_file_spec, MAXLEN - 1);
210 strnz__cpy(file_spec, file_path, MAXLEN - 1);
211 strnz__cat(file_spec, file_name, MAXLEN - 1);
212 } else {
213 tmp_file_spec[last_slash] = '\0';
214 strnz__cpy(file_path, tmp_file_spec, MAXLEN - 1);
215 strnz__cat(file_path, "/", MAXLEN - 1);
216 if (last_slash < file_spec_l)
217 last_slash++;
218 strnz__cpy(file_name, tmp_file_spec + last_slash, MAXLEN - 1);
219 }
220 return (0);
221}
