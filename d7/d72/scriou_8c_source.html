122char di_getch() {
123 struct termios org_tioctl, new_tioctl;
124 char buf;
125
126 if (tcgetattr(2, &org_tioctl) == -1) {
127 fprintf(stderr, "\ndi_getch: tcgetattr failed\n");
128 return (0);
129 }
130 new_tioctl = org_tioctl;
131 new_tioctl.c_lflag &= ~(ECHO | ICANON);
132 new_tioctl.c_cc[VMIN] = 1;
133 new_tioctl.c_cc[VTIME] = 0;
134 tcsetattr(2, TCSAFLUSH, &new_tioctl);
135 read(2, &buf, 1);
136 tcsetattr(2, TCSAFLUSH, &org_tioctl);
137 return (buf);
