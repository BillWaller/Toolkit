99 @note Sets signal handlers to default in the child process.

101 @note Handles errors from fork and execvp, and reports child exit status.
102 @note Restores curses mode and keypad settings after execution.
103 @note Restores window states after execution.
104 @note Uses a temporary string buffer tmp_str for error messages.
105 @note Uses Perror for error reporting.
106 @note Uses sig_dfl_mode and sig_prog_mode for signal handling.
107 @note Uses capture_curses_tioctl and restore_curses_tioctl for terminal
108 settings.
109 @note Uses restore_shell_tioctl for shell terminal settings.
110 @note Uses waitpid to wait for the child process.
111 @note Uses WIFEXITED, WEXITSTATUS, WIFSIGNALED, and WTERMSIG to interpret
112 child status.
113 @note Uses keypad to manage keypad mode in curses.
114 @note Uses restore_wins to restore window states.
115 @note Uses errno for error codes.
116 @note Uses pid_t for process IDs.
117 @note Uses standard file descriptors STDIN_FILENO, STDOUT_FILENO,
118 STDERR_FILENO.
119 @note Uses execvp for executing the command.
120 @note Uses fork for creating a new process.
121 @note Uses ssnprintf for formatting error messages.
122 @note Uses switch-case for handling fork results.
123 @note Uses default shell if SHELL environment variable is not set. */
124int fork_exec(char **argv) {
125 char tmp_str[MAXLEN];
126 pid_t pid;
127 int status;
128 int rc;
129
130 if (argv[0] == 0) {
131 Perror("fork_exec: missing argument for execvp");
132 return (-1);
133 }
134 capture_curses_tioctl();
135 sig_dfl_mode();
136 restore_shell_tioctl();
137
138 tmp_str[0] = '\0';
139 pid = fork();
140 switch (pid) {
141 case -1: // parent fork failed
142 restore_curses_tioctl();
143 sig_prog_mode();
144 keypad(stdscr, true);
145 ssnprintf(tmp_str, sizeof(tmp_str), "fork failed: %s, errno: %d",
146 argv[0], errno);
147 Perror(tmp_str);
148 return (-1);
149 case 0: // child
150 restore_shell_tioctl();
151 execvp(argv[0], argv);
152 restore_curses_tioctl();
153 sig_prog_mode();
154 keypad(stdscr, true);
155 ssnprintf(tmp_str, sizeof(tmp_str), "execvp failed: %s, errno: %d",
156 argv[0], errno);
157 Perror(tmp_str);
158 exit(-1);
159 default: // parent
160 rc = 0;
161 restore_curses_tioctl();
162 sig_prog_mode();
163 waitpid(pid, &status, 0);
164 if (WIFEXITED(status)) {
165 rc = WEXITSTATUS(status);
166 if (rc != 0) {
167 keypad(stdscr, true);
168 ssnprintf(tmp_str, sizeof(tmp_str),
169 "Child %s exited with status %d", argv[0], rc);
170 }
171 } else {
172 if (WIFSIGNALED(status)) {
173 rc = WTERMSIG(status);
174 keypad(stdscr, true);
175 ssnprintf(tmp_str, sizeof(tmp_str),
176 "Child %s terminated by signal %d", argv[0], rc);
177 } else {
178 keypad(stdscr, true);
179 ssnprintf(tmp_str, sizeof(tmp_str),
180 "Child %s terminated abnormally", argv[0]);
181 }
182 }
183 break;
184 }
185 restore_curses_tioctl();
186 sig_prog_mode();
187 keypad(stdscr, true);
188 restore_wins();
189 if (tmp_str[0] != '\0') {
190 Perror(tmp_str);
191 }
192 return (rc);
193}
