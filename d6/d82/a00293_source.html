36/** @brief Accept input for a field
37 @param form Pointer to Form structure
38 @return int Key code of action taken
39 @note Handles character input, navigation keys, and mouse events for field
40 editing. Validates input based on field format and updates display.
41 Returns key code of action taken (e.g., accept field, tab to next field,
42 break out of input loop).
43 */
44int form_accept_field(Form *form) {
45 bool f_insert = FALSE;
46 int in_key;
47 char *s, *d;
48 WINDOW *win = form->win;
49
50 int flin = form->field[form->fidx]->line;
51 int fcol = form->field[form->fidx]->col;
52 int flen = form->field[form->fidx]->len;
53 int ff = form->field[form->fidx]->ff;
54 char *accept_s = form->field[form->fidx]->accept_s;
55 char *filler_s = form->field[form->fidx]->filler_s;
56 form_fmt_field(form, accept_s);
57 click_x = click_y = -1;
58 char *fstart = accept_s;
59 char *fend = fstart + flen;
60 int x = fcol;
61 char *p = fstart = accept_s;
62 char *str_end = p + strlen(p);
63 in_key = 0;
64 if (f_insert)
65 set_chyron_key(form->chyron, 18, "OVR", KEY_IC);
66 else
67 set_chyron_key(form->chyron, 18, "INS", KEY_IC);
68 compile_chyron(form->chyron);
69 form_display_chyron(form);
70
71 while (1) {
72 if (in_key == 0) {
73 mvwaddstr(win, flin, fcol, filler_s);
74 mvwaddstr(win, flin, fcol, accept_s);
75 wmove(win, flin, x);
76 tcflush(0, TCIFLUSH);
77 wmove(win, flin, x);
78 in_key = xwgetch(win, form->chyron);
79 }
80 switch (in_key) {
81 /** KEY_F(10) is the default key for accepting the field and moving
82 to the next field */
83 case KEY_F(10):
84 form_fmt_field(form, accept_s);
85 form_display_field(form);
86 if (form_validate_field(form) != 0)
87 continue;
88 return (in_key);
89 /** KEY_F(9) Cancels the current operation */
90 case KEY_BREAK:
91 case KEY_F(9):
92 form_display_field(form);
93 in_key = KEY_F(9);
94 return (in_key);
95 /** KEY_UP, KEY_BTAB moves to the previous field */
96 case 'H':
97 case KEY_F(1):
98 return (in_key);
99 case KEY_BTAB:
100 case KEY_UP:
101 form_fmt_field(form, accept_s);
102 form_display_field(form);
103 in_key = KEY_UP;
104 return (in_key);
105 /** KEY_DOWN, TAB moves to the next field */
106 case '\t':
107 case KEY_DOWN:
108 form_fmt_field(form, accept_s);
109 form_display_field(form);
110 in_key = KEY_DOWN;
111 return (in_key);
112 /** KEY_END moves cursor to end of field */
113 case KEY_END:
114 while (*p != '\0')
115 p++;
116 x = fcol + (p - fstart);
117 in_key = 0;
118 continue;
119 /**< Enter accepts the field and moves to the next field if
120 * form->f_erase_remainder is set, it will clear the remaining
121 * characters above and after the current cursor location */
122 case '\n':
123 case KEY_ENTER:
124 if (form->f_erase_remainder)
125 *p = '\0';
126 form_fmt_field(form, accept_s);
127 form_display_field(form);
128 in_key = KEY_ENTER;
129 return (in_key);
130 /** KEY_IC toggles insert mode */
131 case KEY_IC:
132 if (f_insert) {
133 f_insert = FALSE;
134 set_chyron_key(form->chyron, 18, "INS", KEY_IC);
135 } else {
136 f_insert = TRUE;
137 set_chyron_key(form->chyron, 18, "OVR", KEY_IC);
138 }
139 compile_chyron(form->chyron);
140 form_display_chyron(form);
141 in_key = 0;
142 continue;
143 /** KEY_DC deletes character at cursor */
144 case KEY_DC:
145 s = p + 1;
146 d = p;
147 while (*s != '\0')
148 *d++ = *s++;
149 *d = '\0';
150 str_end = d;
151 f_insert = FALSE;
152 in_key = 0;
153 continue;
154 /** KEY_HOME moves cursor to start of field */
155 case KEY_HOME:
156 p = fstart;
157 x = fcol;
158 in_key = 0;
159 continue;
160 /** KEY_BACKSPACE deletes character before cursor */
161 case KEY_BACKSPACE:
162 if (p > fstart) {
163 p--;
164 x--;
165 }
166 s = p + 1;
167 d = p;
168 while (*s != '\0')
169 *d++ = *s++;
170 *d = '\0';
171 str_end = d;
172 in_key = 0;
173 continue;
174 /** KEY_LEFT moves cursor left one character */
175 case KEY_LEFT:
176 if (p > fstart) {
177 p--;
178 x--;
179 }
180 in_key = 0;
181 continue;
182 /** KEY_RIGHT moves cursor right one character */
183 case KEY_RIGHT:
184 if (p < fend && p < str_end) {
185 p++;
186 x++;
187 }
188 in_key = 0;
189 continue;
190 /** Handles mouse events for field editing */
191 case KEY_MOUSE:
192 form->fidx = form_yx_to_fidx(form, click_y, click_x);
193 x = click_x;
194 flin = click_y;
195 flen = form->field[form->fidx]->len;
196 ff = form->field[form->fidx]->ff;
197 accept_s = form->field[form->fidx]->accept_s;
198 filler_s = form->field[form->fidx]->filler_s;
199 form_fmt_field(form, accept_s);
200 fstart = accept_s;
201 fend = fstart + flen;
202 p = fstart + (x - form->field[form->fidx]->col);
203 str_end = fstart + strlen(fstart);
204 in_key = 0;
205 continue;
206
207 default:
208 if (p >= fend) {
209 in_key = 0;
210 continue;
211 }
212 /** Validates fields based on format */
213 switch (ff) {
214 /** FF_STRING accepts all printable characters and spaces */
215 case FF_STRING:
216 break;
217 /** FF_DECIMAL_INT accepts digits 0 through 9 and decimal point
218 * ('.') */
219 case FF_DECIMAL_INT:
220 if ((in_key >= '0' && in_key <= '9') || in_key == '.')
221 break;
222 beep();
223 in_key = 0;
224 continue;
225 /** FF_HEX_INT accepts digits 0 through 9 and letters A through
226 * F (case-insensitive) */
227 case FF_HEX_INT:
228 if ((in_key >= '0' && in_key <= '9') ||
229 (in_key >= 'A' && in_key <= 'F') ||
230 (in_key >= 'a' && in_key <= 'f'))
231 break;
232 beep();
233 in_key = 0;
234 continue;
235 /** FF_FLOAT accepts digits 0 through 9 and decimal point ('.')
236 * and negative operator ('-') at the start of the field */
237 case FF_FLOAT:
238 if ((in_key >= '0' && in_key <= '9') || in_key == '.' ||
239 (in_key == '-' && p == fstart))
240 break;
241 beep();
242 in_key = 0;
243 continue;
244 /** FF_DOUBLE accepts digits 0 through 9 and decimal point ('.')
245 * and negative operator ('-') at the start of the field */
246 case FF_DOUBLE:
247 if ((in_key >= '0' && in_key <= '9') || in_key == '.' ||
248 (in_key == '-' && p == fstart))
249 break;
250 beep();
251 in_key = 0;
252 continue;
253 /** FF_CURRENCY accepts digits 0 through 9 and decimal point
254 * ('.') and negative operator ('-') at the start of the field
255 */
256 case FF_CURRENCY:
257 if ((in_key >= '0' && in_key <= '9') || in_key == '.' ||
258 (in_key == '-' && p == fstart))
259 break;
260 beep();
261 in_key = 0;
262 continue;
263 /** FF_YYYYMMDD accepts digits 0 through 9 */
264 case FF_YYYYMMDD:
265 if (in_key >= '0' && in_key <= '9')
266 break;
267 beep();
268 in_key = 0;
269 continue;
270 /** FF_HHMMSS accepts digits 0 through 9 */
271 case FF_HHMMSS:
272 if (in_key >= '0' && in_key <= '9')
273 break;
274 beep();
275 in_key = 0;
276 continue;
277 /** FF_APR accepts digits 0 through 9 and decimal point ('.') */
278 case FF_APR:
279 if ((in_key >= '0' && in_key <= '9') || in_key == '.')
280 break;
281 beep();
282 in_key = 0;
283 continue;
284 default:
285 Perror("form_accept_field() invalid format");
286 break;
287 }
288 if (in_key >= ' ') {
289 if (f_insert) {
290 if (str_end < fend) {
291 s = str_end - 1;
292 d = str_end;
293 while (s >= p)
294 *d-- = *s--;
295 *p++ = in_key;
296 str_end++;
297 x++;
298 }
299 } else {
300 if (p < fend) {
301 if (p < str_end) {
302 *p++ = in_key;
303 x++;
304 } else if (p == str_end) {
305 *p++ = in_key;
306 *p = '\0';
307 str_end = p;
308 x++;
309 }
310 }
311 }
312 in_key = 0;
313 continue;
314 }
315 }
316 }
317}

335/** @brief Display current field

353 return 0;

378/** @brief Format field according to its format type
379 @param form Pointer to Form structure
380 @param s Input string to format
381 @return 0 on success, non-zero on error
382 @note This function takes the input string for the current field and formats
383 it according to the field's specified format type (ff). It updates the
384 accept_s and display_s strings for the field based on the formatted value.
385 The function handles various format types, including strings, decimal
386 integers, hexadecimal integers, floating-point numbers, currency, dates, and
387 times. It uses helper functions for validation and formatting, such as
388 is_valid_date(), is_valid_time(), numeric(), right_justify(), left_justify(),
389 and strnzcpy(). The function ensures that the formatted output fits within
390 the field's length and creates a filler string for the field as needed. The
391 function also handles error cases, such as invalid formats, and provides
392 feedback through error messages. It is designed to be extensible, allowing
393 for additional format types to be added in the future as needed.
394 */
395int form_fmt_field(Form *form, char *s) {
396 /** @brief Format field according to its format type
397 @param form Pointer to Form structure
398 @param s Input string to format
399 @return 0 on success, non-zero on error
400 @note takes the input string for the current field and formats it
401 according to the field's specified format type (ff). It updates the

404 @note handles various format types, including strings, decimal integers,

406 @note uses helper functions for validation and formatting, such as

410 field as needed. The function also handles error cases, such as invalid
411 formats, and provides feedback through error messages. It is designed to

413 future as needed.

417 @note The function currently does not handle localization or

419 separators or currency symbols. Future enhancements may include support
420 for locale-specific formatting. @note Error handling is basic; future
421 versions may include more detailed error reporting and logging
422 mechanisms. @note Performance optimizations may be considered for
423 handling large volumes of data or high-frequency updates in real-time
424 applications. @note The following variables and structures are used in
425 this function:
426 @code
427 char field_s[MAXLEN];
428 int decimal_int_n = 0;
429 int hex_int_n = 0;
430 float float_n = 0.0;
431 double double_n = 0.0;
432 double currency_n = 0.0;
433 struct Date { int yyyy; int mm; int dd; };
434 struct Time { int hh; int mm; int ss; };
435 @endcode
436 @note supports format types: FF_STRING, FF_DECIMAL_INT, FF_HEX_INT,
437 FF_FLOAT, FF_DOUBLE, FF_CURRENCY, FF_YYYYMMDD, FF_HHMMSS, FF_APR with
438 appropriate formatting and validation for each type. @note Handles field
439 length and filler string creation. @note As the roadmap indicates, these
440 data types are just a start and more complex types and formats may be
441 added in the future. @note These data types are not suitable for
442 financial, scientific, or high-precision applications. They are intended

444 @note In the future, we will aqdd support for additional data types and
445 formats such as: dates with time zones, timestamps, percentages,

452 @note C-Menu Form converts each field's input string into internal
453 numeric binary based on the field's specified format. The internal
454 numberic binary is then formated and displayed, verifying the user's
455 input and Form's interpretation of the user's input.
456 */
457 strnz__cpy(form->field[form->fidx]->input_s, s, MAXLEN - 1);
458 char *input_s = form->field[form->fidx]->input_s;
459 char *accept_s = form->field[form->fidx]->accept_s;
460 char *display_s = form->field[form->fidx]->display_s;
461 char *filler_s = form->field[form->fidx]->filler_s;
462 int ff = form->field[form->fidx]->ff;
463 int fl = form->field[form->fidx]->len;
464
465 char field_s[MAXLEN];
466 int decimal_int_n = 0;
467 int hex_int_n = 0;
468 float float_n = 0.0;
469 double double_n = 0.0;
470 double currency_n = 0.0;
471
472 struct {
473 int yyyy;
474 int mm;
475 int dd;
476 } Date;
477 Date.yyyy = Date.mm = Date.dd = 0;
478 struct {
479 int hh;
480 int mm;
481 int ss;
482 } Time;
483 Time.hh = Time.mm = Time.ss = 0;
484 strnz(accept_s, fl);
485 switch (ff) {
486 case FF_STRING:
487 left_justify(s);
488 trim(s);
489 strnz__cpy(input_s, s, MAXLEN - 1);

531 break;
532 case FF_HHMMSS:
533 Time.hh = Time.mm = Time.ss = 0;
534 strnz__cpy(field_s, input_s, MAXLEN - 1);
535 sscanf(field_s, "%2d%2d%2d", &Time.hh, &Time.mm, &Time.ss);
536 sprintf(accept_s, "%02d%02d%02d", Time.hh, Time.mm, Time.ss);

538 sprintf(display_s, "%02d:%02d:%02d", Time.hh, Time.mm, Time.ss);
539 break;
540 case FF_APR:
541 sscanf(input_s, "%lf", &double_n);
542 sprintf(accept_s, "%lf", double_n);
543 sprintf(display_s, "%0.3lf", double_n);
544 right_justify(display_s, fl);
545 break;
546 default:
547 Perror("form_fmt_field() invalid format");
548 break;
549 }
550 strnz(accept_s, fl);
551 left_justify(accept_s);
552 mk_filler(filler_s, fl);
553 return 0;
554}
555/** @brief Validate current field based on flags

563 if (form->field[n]->ff & F_NOTBLANK) {

597/** @brief Left justify string by removing leading spaces
598 @param s String to left justify
599 @note This function takes a string s and removes any leading spaces,
600 effectively left-justifying the text. It does this by finding the first
601 non-space character and shifting the string to the left, overwriting the
602 leading spaces. The resulting string is null-terminated.
603 */
604void left_justify(char *s) { trim(s); }
605/** @brief Right justify string by removing trailing spaces and adding leading
606 spaces
607 @param s String to right justify
608 @param fl Field length
609 @note This function takes a string s and right-justifies it within a field
610 of length fl. It first removes any trailing spaces from the string, then
611 shifts the characters to the right end of the field, filling the left side
612 with spaces. The resulting string is null-terminated and fits within the
613 specified field length.
614 */
615void right_justify(char *s, int fl) {
616 char *p = s;
617 char *d = s + fl;
618 trim(s);
619 *d = '\0';
620 while (*s != '\0') {
621 s++;
622 }
623 while (s != p) {
624 *(--d) = *(--s);
625 }
626 while (d != p) {
627 *(--d) = ' ';
628 }
629}
630/** @brief Check if a given date is valid, including leap years
631 @param yyyy Year
632 @param mm Month
633 @param dd Day
634 @return true if the date is valid, false otherwise
635 @note This function checks if the provided year, month, and day constitute a
636 valid date. It accounts for leap years when determining the number of days in
637 February. The function returns true if the date is valid and false if it is
638 not.
639 */
640bool is_valid_date(int yyyy, int mm, int dd) {
641 if (yyyy < 1 || mm < 1 || mm > 12 || dd < 1)
642 return false;
643 int days_in_month[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
644 if ((yyyy % 4 == 0 && yyyy % 100 != 0) || (yyyy % 400 == 0))
645 days_in_month[2] = 29;
646 if (dd > days_in_month[mm])
647 return false;
648 return true;
649}
650/** @brief Check if a given time is valid
651 @param hh Hour
652 @param mm Minute
653 @param ss Second
654 @return true if the time is valid, false otherwise
655 @note This function checks if the provided hour, minute, and second
656 constitute a valid time. It ensures that hours are between 0 and 23, minutes
657 and seconds are between 0 and 59. The function returns true if the time is
658 valid and false if it is not.
