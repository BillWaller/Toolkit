53int init_form(Init *init, int argc, char **argv, int begy, int begx) {

89/** @brief Form main processing loop
90 @param init A pointer to the Init structure containing form data and state.
91 @return 0 on successful completion, or a non-zero value if the user cancels
92 the form or if an error occurs during processing.
93 1. Parse the form description file to populate the form data structure.
94 2. Read any initial data for the form fields from a specified input
95 source.
96 3. Display the form on the screen with the initial field values.
97 4. Enter a loop to handle user input for field entry, calculation, help,
98 and cancellation:
99 a. If the user selects the accept action, perform any necessary
100 calculations or post-processing, and then either return to field entry
101 or exit the loop if the form is accepted.
102 b. If the user selects the help action, display the help screen and
103 return to the form after the user exits the help screen.
104 c. If the user selects the cancel action, exit the loop and return a
105 cancel status. */
106int form_engine(Init *init) {
107 char tmp_str[MAXLEN];
108 int form_action;
109
110 form = init->form;
111 if (form == nullptr) {
112 Perror("FORM: form data structure is nullptr");
113 }
114 if (form_parse_desc(form)) {
115 destroy_form(init);
116 return 0;
117 }
118 form_read_data(form);
119 form_display_screen(init);
120 form->fidx = 0;
121 form_action = 0;
122 while (1) {
123 if (form_action == 0 || form_action == P_CONTINUE)
124 form_action = form_enter_fields(form);
125 switch (form_action) {
126 case P_ACCEPT:
127 if (form->f_calculate)
128 form_action = form_calculate(init);
129 else
130 form_action = form_end_fields(init);
131 if (form_action == P_HELP || form_action == P_CANCEL ||
132 form_action == P_CONTINUE || form_action == P_END)
133 continue;
134 if (form_action == P_ACCEPT) {
135 form_action = P_END;
136 continue;
137 }
138 break;
139 case P_END:
140 if (form->f_out_spec || form->out_spec[0] != '\0')
141 form_write(form);
142 if (form->f_receiver_cmd) {
143 form_exec_cmd(form);
144 form_action = P_CONTINUE;
145 continue;
146 }
147 return 0;
148 case P_HELP:
149 eargv[0] = strdup("mview");
150 strnz__cpy(tmp_str, "~/menuapp/help/form.help", MAXLEN - 1);
151 expand_tilde(tmp_str, MAXLEN - 1);
152 eargv[1] = strdup(tmp_str);
153 eargv[2] = nullptr;
154 eargc = 2;
155 zero_opt_args(init);
156 parse_opt_args(init, eargc, eargv);
157 init->lines = 30;
158 init->cols = 60;
159 init->begy = menu->begy + 1;
160 init->begx = menu->begx + 4;
161 strnz__cpy(init->title, "Form Help", MAXLEN - 1);
162 mview(init, eargc, eargv);
163 restore_wins();
164 form_action = P_CONTINUE;
165 break;
166 case P_CANCEL:
167 return P_CANCEL;
168 default:
169 form_action = P_CONTINUE;
170 break;
171 }
172 }
173 return 0;
174}

226/** @brief Handle integration with an external program for calculation, allowing
227 the user to execute a provider command and read results back into the form
228 fields.
229 @param init A pointer to the Init structure containing form data and state.
230 @return An integer status code indicating the next action for the form
231 processing loop (e.g., P_CONTINUE, P_CANCEL, P_ACCEPT, P_HELP).
232 @details This function provides integration with external programs.
233 The requirements are:
234 1. The form description file must have a line starting with 'C' to
235 indicate that the form supports calculation.
236 2. The form description file must specify the provider command using a
237 line starting with '!' followed by the command and its arguments.
238 3. The external program must be able to accept field data from a file,
239 from standard input, or as command line parameters.
240 4. The external program must output the calculated field values in a
241 format that can be read by the form (e.g., one value per line), either
242 to a file or to standard output.
243
244 The sequence of operations is as follows:
245
246 1. The 'C' option causes Form to pause and display an KEY_F(5) Calculate
247 option on the chyron.
248 2. The user can then cancel the operation by pressing KEY_F(9) or
249 activate the calculate option by pressing KEY_F(5).
250 3. Form outputs its data to file, standard output, or as command line
251 parameters.
252 4. Form executes the external program.
253 5. The external program processes the data and outputs the results.
254 6. Form reads the results and populates the appropriate form fields.
255 7. Form presents the user with an option to edit the data, and the
256 sequence restarts at 1.
257
258 This function forks and executes the provider executable as a child
259 process, creates a pipe to read the output from the provider command,
260 reads the output, and updates the Form fields.
261 */
262int form_calculate(Init *init) {
263 int i, c, rc;
264 char earg_str[MAXLEN + 1];
265 char *eargv[MAXARGS];
266 char file_spec[MAXLEN];
267 bool loop = true;
268 pid_t pid;
269 int pipe_fd[2];
270
271 form = init->form;
272 wmove(form->win, form->lines - 1, 0);
273 wclrtoeol(form->win);
274 unset_chyron_key(form->chyron, 18);
275 unset_chyron_key(form->chyron, 10);
276 set_chyron_key(form->chyron, 5, "F5 Calculate", KEY_F(5));
277
278 while (loop) {
279 form_display_chyron(form);
280 click_y = click_x = -1;
281 tcflush(2, TCIFLUSH);
282 c = xwgetch(form->win, form->chyron);
283 switch (c) {
284 case KEY_F(1):
285 return P_HELP;
286 case KEY_F(5):
287 if (form->f_out_spec)
288 form_write(form);
289 if (form->f_provider_cmd) {
290 strnz__cpy(earg_str, form->provider_cmd, MAXLEN - 1);
291 for (i = 0; i < form->fcnt; i++) {
292 strnz__cat(earg_str, " ", MAXLEN - 1);
293 strnz__cat(earg_str, form->field[i]->accept_s, MAXLEN - 1);
294 }
295 str_to_args(eargv, earg_str, MAXARGS);
296 strnz__cpy(file_spec, eargv[0], MAXLEN - 1);
297 base_name(eargv[0], file_spec);
298 if (pipe(pipe_fd) == -1) {
299 Perror("pipe(pipe_fd) failed in init_form");
300 return (1);
301 }
302 if ((pid = fork()) == -1) {
303 Perror("fork() failed in init_form");
304 return (1);
305 }
306 if (pid == 0) { // Child
307 close(pipe_fd[P_READ]);
308 dup2(pipe_fd[P_WRITE], STDOUT_FILENO);
309 close(pipe_fd[P_WRITE]);
310 execvp(eargv[0], eargv);
311 ssnprintf(em0, MAXLEN, "%s, line: %d", __FILE__,
312 __LINE__ - 2);
313 strnz__cpy(em1, "execvp(", MAXLEN - 1);
314 strnz__cat(em1, eargv[0], MAXLEN - 1);
315 strnz__cat(em1, ", ", MAXLEN - 1);
316 strnz__cat(em1, earg_str, MAXLEN - 1);
317 strnz__cat(em1, ")", MAXLEN - 1);
318 strerror_r(errno, em2, MAXLEN);
319 display_error(em0, em1, em2, nullptr);
320 exit(EXIT_FAILURE);
321 } // Back to parent
322 close(pipe_fd[P_WRITE]);
323 form->in_fp = fdopen(pipe_fd[P_READ], "rb");
324 form->f_in_pipe = true;
325 form_read_data(form);
326 close(pipe_fd[P_READ]);
327 waitpid(pid, nullptr, 0);
328 form_display_fields(form);
329 set_chyron_key(form->chyron, 8, "F5 Edit", KEY_F(5));
330 set_chyron_key(form->chyron, 10, "F10 Commit", KEY_F(10));
331 continue;
332 }
333 break;
334 case KEY_F(8):
335 if (is_set_chyron_key(form->chyron, 8)) {
336 loop = false;
337 rc = P_CONTINUE;
338 break;
339 }
340 continue;
341 case KEY_F(9):
342 loop = false;
343 rc = P_CANCEL;
344 break;
345 case KEY_F(10):
346 loop = false;
347 rc = P_ACCEPT;
348 break;
349 case KEY_MOUSE:
350 break;
351 default:
352 break;
353 }
354 }
355 unset_chyron_key(form->chyron, 8);
356 unset_chyron_key(form->chyron, 5);
357 form_display_chyron(form);
358 return rc;
359}
360/** @brief Handle user input for field entry, allowing navigation between fields
361 and looping until an exit action is selected.
362 @param form A pointer to the Form structure containing form data and state.
363 @return An integer status code indicating the next action for the form
364 processing loop (e.g., P_ACCEPT, P_HELP, P_CALC, P_CANCEL).
365 @note This function manages user input for field entry, including navigation
366 between fields and handling of special keys for accepting, canceling,
367 requesting help, or performing calculations. The function loops until the
368 user selects an exit action (e.g., accept or cancel). */
369int form_enter_fields(Form *form) {
370
371 if (form->fidx < 0)
372 return (-1);
373 while (1) {
374 cmd_key = form_accept_field(form);
375 wrefresh(form->win);
376
377 switch (cmd_key) {
378
379 case KEY_F(10):
380 return (P_ACCEPT);
381
382 case KEY_F(1):
383 return (P_HELP);
384
385 case KEY_F(5):
386 if (form->f_calculate)
387 return (P_CALC);
388 break;
389
390 case KEY_F(9):
391 return (P_CANCEL);
392
393 case KEY_UP:
394 if (form->fidx != 0)
395 form->fidx--;
396 break;
397
398 case '\r':
399 case KEY_ENTER:
400 if (form->fidx < form->fcnt - 1)
401 form->fidx++;
402 else if (form->fidx == form->fcnt - 1)
403 return (P_ACCEPT);
404 break;
405
406 case KEY_DOWN:
407 if (form->fidx < form->fcnt - 1)
408 form->fidx++;
409 else if (form->fidx == form->fcnt - 1)
410 return (P_ACCEPT);
411 break;
412 case KEY_MOUSE:
413 break;
414 default:
415 break;
416 }
417 }
418}

517/** @brief Parse the form description file to populate the Form data structure
518 with field definitions, text elements, and other configuration specified in
519 the description file.
520 @param form A pointer to the Form structure containing form data and state.
521 @return 0 on success, or a non-zero value if an error occurs while parsing
522 the description file (e.g., file not found, invalid format, missing
523 directives). */
524int form_parse_desc(Form *form) {
525 FILE *form_desc_fp;
526 char *token;
527 char *s;
528 int cols = 0;
529 int i, l;
530 int in_line_num = 0;
531 char in_buf[BUFSIZ];
532 char tmp_buf[BUFSIZ];
533 char *tmp_buf_p;
534 char tmp_str[BUFSIZ];
535 char delim[5];
536 char directive;
537
538 form_desc_fp = fopen(form->mapp_spec, "r");
539 if (form_desc_fp == nullptr) {
540 ssnprintf(em0, MAXLEN - 1, "%s, line: %d", __FILE__, __LINE__ - 2);
541 strnz__cpy(em1, "fopen ", MAXLEN - 1);
542 strnz__cat(em1, form->mapp_spec, MAXLEN - 1);
543 strerror_r(errno, em2, MAXLEN);
544 display_error(em0, em1, em2, nullptr);
545 return (1);
546 }
547 for (i = 0; i < MAXFIELDS; i++) {
548 form->field[i] = (Field *)calloc(1, sizeof(Field));
549 if (!form->field[i]) {
550 sprintf(tmp_str, "FORM: calloc failed for fields");
551 abend(EXIT_FAILURE, tmp_str);
552 }
553 }
554 for (i = 0; i < MAXFIELDS; i++) {
555 form->text[i] = (Text *)calloc(1, sizeof(Text));
556 if (!form->field[i]) {
557 sprintf(tmp_str, "FORM: calloc failed for text");
558 abend(EXIT_FAILURE, tmp_str);
559 }
560 }
561 form->didx = 0;
562 form->fidx = 0;
563 form->fcnt = 0;
564 form->cols = 34;
565 while ((fgets(in_buf, MAXLEN, form_desc_fp)) != nullptr) {
566 s = in_buf;
567 in_line_num++;
568 l = trim(in_buf);
569 if (l == 0)
570 continue;
571 if (*s == D_COMMENT)
572 continue;
573 delim[0] = '\n';
574 delim[1] = in_buf[1];
575 delim[2] = '\0';
576 strnz__cpy(tmp_buf, in_buf, MAXLEN - 1);
577 tmp_buf_p = tmp_buf;
578 if (!(token = strtok(tmp_buf_p, delim))) {
579 continue;
580 }
581 directive = *token;
582 switch ((int)directive) {
583 case D_COMMENT:
584 break;
585 case D_CALC:
586 form->f_calculate = true;
587 break;
588 case D_QUERY:
589 form->f_query = true;
590 break;
591 case D_CMD:
592 if (!(token = strtok(nullptr, delim))) {
593 form_desc_error(in_line_num, in_buf,
594 "FORM: receiver_cmd delimiter");
595 continue;
596 }
597 strnz__cpy(form->receiver_cmd, token, MAXLEN - 1);
598 break;
599 case D_HELP:
600 if (!(token = strtok(nullptr, delim))) {
601 form_desc_error(in_line_num, in_buf,
602 "FORM: help_spec delimiter");
603 }
604 strnz__cpy(form->help_spec, token, MAXLEN - 1);
605 break;
606 case D_FIELD:
607 if (form->field[form->fidx] == nullptr) {
608 sprintf(tmp_str, "FORM: calloc failed for fields");
609 abend(EXIT_FAILURE, tmp_str);
610 }
611 if (!(token = strtok(nullptr, delim))) {
612 form_desc_error(in_line_num, in_buf,
613 "FORM: line number delimiter");
614 return 1;
615 }
616 form->field[form->fidx]->line = atoi(token);
617 if (form->field[form->fidx]->line < 0 ||
618 form->field[form->fidx]->line >= MAXFIELDS) {
619 form_desc_error(in_line_num, in_buf,
620 "FORM: invalid line number");
621 return 1;
622 }
623 if (!(token = strtok(nullptr, delim))) {
624 form_desc_error(in_line_num, in_buf,
625 "FORM: column number delimiter");
626 return 1;
627 }
628 form->field[form->fidx]->col = atoi(token);
629 if (form->field[form->fidx]->col < 0 ||
630 form->field[form->fidx]->col >= TBL_COLS) {
631 form_desc_error(in_line_num, in_buf,
632 "FORM: invalid column number");
633 break;
634 }
635 if (!(token = strtok(nullptr, delim))) {
636 strnz__cpy(tmp_str, in_buf, MAXLEN - 1);
637 form_desc_error(in_line_num, tmp_str, "FORM: length delimiter");
638 break;
639 }
640 form->field[form->fidx]->len = atoi(token);
641 if (form->field[form->fidx]->len < 0 ||
642 form->field[form->fidx]->len > TBL_COLS) {
643 form_desc_error(in_line_num, in_buf, "FORM: invalid length");
644 break;
645 }
646 if (!(token = strtok(nullptr, delim))) {
647 form_desc_error(in_line_num, in_buf,
648 "FORM: validation code delimiter");
649 break;
650 }
651 form->field[form->fidx]->ff = -1;
652 for (i = 0; i < FF_INVALID; i++) {
653 str_to_lower(token);
654 str_to_lower(ff_tbl[i]);
655 if (!strcmp(token, ff_tbl[i])) {
656 form->field[form->fidx]->ff = i;
657 break;
