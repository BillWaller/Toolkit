42/** @brief Initializes pick structure and opens pick input file or pipe
43 @param init Pointer to Init structure
44 @param argc Argument count
45 @param argv Argument vector
46 @param begy Beginning y coordinate for pick window
47 @param begx Beginning x coordinate for pick window
48 @note If provider_cmd is specified, it takes precedence over in_spec and
49 input file arguments
50 @note provider_cmd is executed and its output is read as pick input
51 @note If provider_cmd is not specified, in_spec is used to read pick
52 input from a file or stdin
53 @note If provider_cmd is specified, it is executed and its output is
54 read as pick input */
55int init_pick(Init *init, int argc, char **argv, int begy, int begx) {
56 struct stat sb;
57 char *s_argv[MAXARGS];
58 char tmp_str[MAXLEN];
59 int m;
60 pid_t pid = 0;
61
62 if (init->pick != nullptr)
63 destroy_pick(init);
64 Pick *pick = new_pick(init, argc, argv, begy, begx);
65 if (init->pick != pick)
66 abend(-1, "init->pick != pick\n");
67 SIO *sio = init->sio;
68 if (pick->provider_cmd[0] != '\0') {
69 str_to_args(s_argv, pick->provider_cmd, MAXARGS - 1);
70 if (pipe(pipe_fd) == -1) {
71 Perror("pipe(pipe_fd) failed in init_pick");
72 return (1);
73 }
74 if ((pid = fork()) == -1) {
75 Perror("fork() failed in init_pick");
76 return (1);
77 }
78 if (pid == 0) {
79 /** Spawn Child to execute provider_cmd
80 Close read end of pipe as Child only needs to write to pipe */
81 close(pipe_fd[P_READ]);
82 /** Connect CHILD STDOUT to write end of pipe */
83 dup2(pipe_fd[P_WRITE], STDOUT_FILENO);
84 dup2(pipe_fd[P_WRITE], STDERR_FILENO);
85 /** STDOUT attached to write end of pipe, so close pipe fd */
86 close(pipe_fd[P_WRITE]);
87 execvp(s_argv[0], s_argv);
88 m = MAXLEN - 24;
89 strnz__cpy(tmp_str, "Can't exec pick start cmd: ", m);
90 m -= strlen(s_argv[0]);
91 strnz__cat(tmp_str, s_argv[0], m);
92 Perror(tmp_str);
93 exit(EXIT_FAILURE);
94 }
95 /** Return to Parent
96 Close write end of pipe as Parent only needs to read from pipe */
97 close(pipe_fd[P_WRITE]);
98 /** Open a file pointer on read end of pipe */
99 pick->in_fp = fdopen(pipe_fd[P_READ], "rb");
100 pick->f_in_pipe = true;
101 } else {
102 if ((pick->in_spec[0] == '\0') || strcmp(pick->in_spec, "-") == 0 ||
103 strcmp(pick->in_spec, "/dev/stdin") == 0) {
104 strnz__cpy(pick->in_spec, "/dev/stdin", MAXLEN - 1);
105 pick->in_fp = fdopen(STDIN_FILENO, "rb");
106 pick->f_in_pipe = true;
107 }
108 }
109 if (!pick->f_in_pipe) {
110 /** No provider_cmd specified, so read pick input from file or stdin */
111 if (lstat(pick->in_spec, &sb) == -1) {
112 m = MAXLEN - 29;
113 strnz__cpy(tmp_str, "Can\'t stat pick input file: ", m);
114 m -= strlen(pick->in_spec);
115 strnz__cat(tmp_str, pick->in_spec, m);
116 Perror(tmp_str);
117 return (1);
118 }
119 if (sb.st_size == 0) {
120 m = MAXLEN - 24;
121 strnz__cpy(tmp_str, "Pick input file empty: ", m);
122 m -= strlen(pick->in_spec);
123 strnz__cat(tmp_str, pick->in_spec, m);
124 Perror(tmp_str);
125 return (1);
126 }
127 if ((pick->in_fp = fopen(pick->in_spec, "rb")) == nullptr) {
128 m = MAXLEN - 29;
129 strnz__cpy(tmp_str, "Can't open pick input file: ", m);
130 m -= strlen(pick->in_spec);
131 strnz__cat(tmp_str, pick->in_spec, m);
132 Perror(tmp_str);
133 return (1);
134 }
135 }
136 read_pick_input(init);
137 if (pick->f_in_pipe && pid > 0) {
138 /** Wait for provider_cmd child process to finish before proceeding */
139 waitpid(pid, nullptr, 0);
140 close(pipe_fd[P_READ]);
141 dup2(sio->stdin_fd, STDIN_FILENO);
142 dup2(sio->stdout_fd, STDOUT_FILENO);
143 restore_curses_tioctl();
144 sig_prog_mode();
145 keypad(pick->win, true);
146 }
147 if (pick->obj_cnt == 0) {
148 Perror("No pick objects available");
149 destroy_pick(init);
150 return (1);
151 }
152 /** Enter pick_engine */
153 pick_engine(init);
154 win_del();
155 destroy_pick(init);
156 return 0;
157}
158/** @brief Reads pick input from file pointer and saves objects into pick
159 structure
160 @param init Pointer to Init structure containing pick information
161 @return 0 on success, -1 if no objects were read
162 @note Reads lines from pick->in_fp and saves them as objects in the pick
163 structure using save_object function. If no objects are read, returns -1.
164 Otherwise, sets obj_cnt to the number of objects read and resets obj_idx to 0
165 before returning 0. */
166int read_pick_input(Init *init) {
167 int i;
168
169 Pick *pick = init->pick;
170 pick->select_cnt = 0;
171 pick->obj_cnt = pick->pg_lines = pick->tbl_cols = 0;
172 pick->obj_idx = pick->tbl_page = pick->y = pick->tbl_col = pick->x = 0;
173 pick->tbl_pages = 1;
174
175 if (pick->in_fp) {
176 while (fgets(pick->in_buf, sizeof(pick->in_buf), pick->in_fp) !=
177 nullptr)
178 save_object(pick, pick->in_buf);
179 } else
180 for (i = 1; i < pick->argc; i++)
181 save_object(pick, pick->argv[i]);
182 if (pick->in_fp != nullptr)
183 fclose(pick->in_fp);
184 if (!pick->obj_idx)
185 return (-1);
186 pick->obj_cnt = pick->obj_idx;
187 pick->obj_idx = 0;
188 return 0;
189}
190/** @brief Initializes pick interface, calculates window size and position, and
191 enters picker loop
192 @param init Pointer to Init structure containing pick information
193 @return Count of selected objects on success, -1 if user cancels
194 @note Initializes key command strings for chyron display and calculates pick
195 window size and position based on terminal size and pick parameters.
196 @note Opens pick window and displays first page of objects. Enters picker
197 loop to handle user input and interactions. If user cancels selection,
198 returns -1.
199 @note If user accepts selection, returns count of selected objects. */
200int pick_engine(Init *init) {
201 /** Initialize window and data structures */
202 int rc;
203 int maxy, maxx, win_maxy, win_maxx;
204 int tbl_max_cols, pg_max_objs;
205 /** Initialize key command strings for chyron display */
206
207 getmaxyx(stdscr, maxy, maxx);
208 /** Calculate pick window size and position based on terminal size and pick
209 parameters */
210 win_maxy = (maxy * 8) / 10;
211 if (win_maxy > (maxy - pick->begy) - 2)
212 win_maxy = (maxy - pick->begy) - 2;
213 win_maxx = (maxx * 9) / 10;
214 if (win_maxx > (maxx - pick->begx) - 2)
215 win_maxx = (maxx - pick->begx) - 2;
216 if (pick->tbl_col_width < 4)
217 pick->tbl_col_width = 4;
218 if (pick->tbl_col_width > win_maxx - 2)
219 pick->tbl_col_width = win_maxx - 2;
220 // populate the columns in sequence
221 if (pick->obj_cnt <= win_maxy) {
222 pick->tbl_lines = pick->obj_cnt;
223 pick->tbl_cols = 1;
224 } else {
225 tbl_max_cols = (win_maxx / (pick->tbl_col_width + 1));
226 pg_max_objs = win_maxy * tbl_max_cols;
227 if (pick->obj_cnt > pg_max_objs)
228 pick->tbl_cols = tbl_max_cols;
229 else
230 pick->tbl_cols = pick->obj_cnt / win_maxy;
231 pick->tbl_lines = pick->obj_cnt / tbl_max_cols;
232 }
233 pick->tbl_pages = (pick->tbl_lines / (win_maxy - 1)) + 1;
234 pick->pg_lines = (pick->tbl_lines / pick->tbl_pages);
235 pick->win_lines = pick->pg_lines + 1;
236 pick->tbl_page = 0;
237 if (pick->begy == 0)
238 pick->begy = (LINES - pick->win_lines) / 5;
239 else if (pick->begy + pick->win_lines > LINES - 4)
240 pick->begy = LINES - pick->win_lines - 2;
241 pick->win_width = (pick->tbl_col_width + 1) * pick->tbl_cols;
242
243 pick->chyron = new_chyron();
244 set_chyron_key(pick->chyron, 1, "F1 Help", KEY_F(1));
245 set_chyron_key(pick->chyron, 9, "F9 Cancel", KEY_F(9));
246 set_chyron_key(pick->chyron, 10, "F10 Continue", KEY_F(10));
247 if (pick->tbl_pages > 1) {
248 set_chyron_key(pick->chyron, 11, "PgUp", KEY_PPAGE);
249 set_chyron_key(pick->chyron, 12, "PgDn", KEY_NPAGE);
250 }
251 set_chyron_key(pick->chyron, 13, "Toggle", 32);
252 compile_chyron(pick->chyron);
253 if (pick->chyron->l > win_maxx)
254 pick->chyron->l = strnz(pick->chyron->s, win_maxx);
255 if (pick->win_width < pick->chyron->l)
256 pick->win_width = pick->chyron->l;
257
258 if (pick->begx + pick->win_width > COLS - 4)
259 pick->begx = COLS - pick->win_width - 2;
260 else if (pick->begx == 0)
261 pick->begx = (COLS - pick->win_width) / 2;
262
263 rc = open_pick_win(init);
264 if (rc)
265 return (rc);
266 /** Enter picker loop to handle user input and interactions */
267 do {
268 display_page(pick);
269 reverse_object(pick);
270 pick->obj_idx = 0;
271 pick->x = 1;
272 mousemask(BUTTON1_CLICKED | BUTTON1_DOUBLE_CLICKED, nullptr);
273 rc = picker(init);
274 if (rc == -1)
275 break;
276 else {
277 if (pick->select_cnt > 0) {
278 if (pick->f_out_spec && pick->out_spec[0])
279 output_objects(pick);
280 if (pick->f_cmd && pick->cmd[0])
281 exec_objects(init);
282 }
283 }
284 deselect_object(pick);
285 } while (1);
286 return (rc);
287}
288/** @brief Saves a string as an object in the pick structure
289 @param pick Pointer to Pick structure
290 @param s String to save as an object
291 @note If the current object index is less than the maximum allowed, saves
292 the string as an object in the pick structure. Updates the column width if
293 necessary and marks the object as not selected. Increments the object index
294 for the next object to be saved. */
295void save_object(Pick *pick, char *s) {
296 int l;
297
298 if (pick->obj_idx < OBJ_MAXCNT - 1) {
299 l = strlen(s);
300 if (l > OBJ_MAXLEN - 1)
301 s[OBJ_MAXLEN - 1] = '\0';
302 if (l > pick->tbl_col_width)
303 pick->tbl_col_width = l;
304 if (l < 1)
305 l = 1;
306 pick->object[pick->obj_idx] = (char *)calloc(l + 1, sizeof(char));
307 strnz__cpy(pick->object[pick->obj_idx], s, l);
308 pick->f_selected[pick->obj_idx] = FALSE;
309 pick->obj_idx++;
310 }
311}

313/** @brief Main loop to handle user input and interactions for pick interface
314 @param init Pointer to Init structure
315 @return Number of selected objects or -1 if user cancels
316 @note Handles user input for navigating and selecting objects in the pick
317 interface.
318 @note Supports various key commands for navigation, selection, and
319 accepting/canceling the selection.
320 @note Updates the display accordingly based on user interactions.
321 @note If the user cancels the selection, returns -1.
322 @note If the user accepts the selection, returns the count of selected
323 objects. */
324int picker(Init *init) {
325 int cmd_key;
326 int display_tbl_page;
327
328 pick = init->pick;
329 click_y = click_x = -1;
330 cmd_key = 0;
331 while (1) {
332 tcflush(tty_fd, TCIFLUSH);
333 if (cmd_key == 0)
334 cmd_key = xwgetch(pick->win, pick->chyron);
335 switch (
336 cmd_key) { /** 'q', or KEY_F(9) cancel selection and exit picker */
337 case 'q':
338 case 'Q':
339 case KEY_F(9):
340 deselect_object(pick);
341 return -1;
342 /** KEY_F(1) or 'H' Displays help screen for pick interface */
343 case KEY_F(1):
344 case 'H':
345 display_pick_help(init);
346 display_page(pick);
347 reverse_object(pick);
348 break;
349 /** 't' or Space Toggles selection of current object */
350 case ' ':
351 case 't':
352 case 'T':
353 toggle_object(pick);
354 if (pick->select_cnt == pick->select_max)
355 return pick->select_cnt;
356 break;
357 /** Enter or KEY_F(10) Accepts current selection and exits picker,
358 * returning count of selected objects */
359 case KEY_F(10):
360 case '\n':
361 case KEY_ENTER:
362 return pick->select_cnt;
363 /** KEY_END Moves selection to last object in list */
364 case KEY_END:
365 mvwaddstr_fill(pick->win, pick->y, pick->x,
366 pick->object[pick->obj_idx], pick->tbl_col_width);
367 display_tbl_page = pick->tbl_page;
368 pick->obj_idx = pick->obj_cnt - 1;
369 pick->tbl_page = pick->obj_idx / (pick->pg_lines * pick->tbl_cols);
370 pick->tbl_line = (pick->obj_idx / pick->tbl_cols) % pick->pg_lines;
371 pick->tbl_col = pick->obj_idx % pick->tbl_cols;
372 pick->y = pick->tbl_line;
373 if (display_tbl_page != pick->tbl_page) {
374 display_page(pick);
375 }
376 reverse_object(pick);
377 break;
378 /** 'l' or KEY_RIGHT Moves selection to next object in list */
379 case 'l':
380 case KEY_RIGHT:
381 mvwaddstr_fill(pick->win, pick->y, pick->x,
382 pick->object[pick->obj_idx], pick->tbl_col_width);
383 /** pick->obj_idx += pick->tbl_lines -> next column */
384 if (pick->tbl_page * pick->pg_lines * pick->tbl_cols +
385 (pick->tbl_col + 1) * pick->pg_lines + pick->tbl_line <
386 pick->obj_cnt - 1 &&
387 pick->tbl_col < pick->tbl_cols - 1)
388 pick->tbl_col++;
389 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
390 pick->tbl_col * pick->pg_lines + pick->tbl_line;
391 reverse_object(pick);
392 break;
393 /** 'h' or KEY_LEFT or Backspace Moves selection to previous object
394 * in list */
395 case 'h':
396 case KEY_LEFT:
397 case KEY_BACKSPACE:
398 mvwaddstr_fill(pick->win, pick->y, pick->x,
399 pick->object[pick->obj_idx], pick->tbl_col_width);
400 if (pick->tbl_col > 0)
401 pick->tbl_col--;
402 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
403 pick->tbl_col * pick->pg_lines + pick->tbl_line;
404 reverse_object(pick);
405 break;
406 /** 'j' or KEY_DOWN Moves selection to next object in list, 'k' or
407 * KEY_UP Moves selection to previous object in list */
408 case 'j':
409 case KEY_DOWN:
410 mvwaddstr_fill(pick->win, pick->y, pick->x,
411 pick->object[pick->obj_idx], pick->tbl_col_width);
412 if (pick->tbl_page * pick->pg_lines * pick->tbl_cols +
413 pick->tbl_col * pick->pg_lines + pick->tbl_line <
414 pick->obj_cnt - 1 &&
415 pick->tbl_line < pick->pg_lines - 1)
416 pick->tbl_line++;
417 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
418 pick->tbl_col * pick->pg_lines + pick->tbl_line;
419 reverse_object(pick);
420 break;
421 /** 'k' or KEY_UP Moves selection to previous object in list */
422 case 'k':
423 case KEY_UP:
424 mvwaddstr_fill(pick->win, pick->y, pick->x,
425 pick->object[pick->obj_idx], pick->tbl_col_width);
426 if (pick->tbl_line > 0)
427 pick->tbl_line--;
428 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
429 pick->tbl_col * pick->pg_lines + pick->tbl_line;
430 reverse_object(pick);
431 break;
432 /** KEY_NPAGE or 'Ctrl+f' Moves selection to next page of objects, */
433 case KEY_NPAGE:
434 case '\06':
435 if (pick->tbl_pages == 1)
436 break;
437 if (pick->tbl_page < pick->tbl_pages - 1) {
438 pick->tbl_page++;
439 pick->pg_line = 0;
440 pick->tbl_col = 0;
441 }
442 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
443 pick->tbl_cols * pick->pg_line + pick->tbl_col;
444 display_page(pick);
445 reverse_object(pick);
446 break;
447 /** KEY_PPAGE or 'Ctrl+b' Moves selection to previous page of
448 * objects */
449 case KEY_PPAGE:
450 case '\02':
451 if (pick->tbl_pages == 1)
452 break;
453 if (pick->tbl_page > 0)
454 pick->tbl_page--;
455 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
456 pick->tbl_cols * pick->pg_line + pick->tbl_col;
457 display_page(pick);
458 reverse_object(pick);
459 break;
460 /** KEY_HOME Moves selection to first object in list */
461 case KEY_HOME:
462 pick->tbl_page = 0;
463 pick->tbl_line = 0;
464 pick->tbl_col = 0;
465 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
466 pick->tbl_cols * pick->pg_line + pick->tbl_col;
467 display_page(pick);
468 reverse_object(pick);
469 break;
470 /** KEY_LL (lower left of numeric pad) Moves selection to last
471 object in list */
472 case KEY_LL:
473 pick->tbl_page = pick->tbl_pages - 1;
474 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
475 pick->tbl_cols * pick->pg_line + pick->tbl_col;
476 display_page(pick);
477 reverse_object(pick);
478 break;
479 /** KEY_MOUSE Handles mouse events for selection and chyron key
480 * activation */
481 case KEY_MOUSE:
482 if (click_y == -1 || click_x == -1)
483 break;
484 unreverse_object(pick);
485 pick->y = click_y;
486 pick->tbl_col = (click_x - 1) / (pick->tbl_col_width + 1);
487 if (pick->tbl_col < 0 || pick->tbl_col >= pick->tbl_cols)
488 break;
489 pick->obj_idx = pick->tbl_page * pick->pg_lines * pick->tbl_cols +
490 pick->tbl_col * pick->pg_lines + pick->y;
491 toggle_object(pick);
492 reverse_object(pick);
493 if (pick->select_cnt == pick->select_max)
494 return pick->select_cnt;
495 wrefresh(pick->win);
496 click_y = click_x = -1;
497 break;
498 default:
499 break;
500 }
501 cmd_key = 0;
502 }
503 return 0;
504}
505/** @brief Displays current page of objects in pick window
506 @param pick Pointer to Pick structure containing objects and display
507 information
508 @note Clears the pick window and displays the current page of objects based
509 on the current table page, line, and column. Marks selected objects with an
510 asterisk. Updates the chyron with page information at the bottom of the pick
511 window. */
512void display_page(Pick *pick) {
513 int y, col, pidx;
514 for (y = 0; y < pick->pg_lines; y++) {
515 wmove(pick->win, y, 0);
516 wclrtoeol(pick->win);
517 }
518 pidx = pick->tbl_page * pick->pg_lines * pick->tbl_cols;
519 for (col = 0; col < pick->tbl_cols; col++) {
520 pick->x = col * (pick->tbl_col_width + 1) + 1;
521 for (y = 0; y < pick->pg_lines; y++, pidx++) {
522 if (pidx < pick->obj_cnt) {
523 if (pick->f_selected[pidx])
524 mvwaddstr(pick->win, y, pick->x - 1, "*");
525 mvwaddstr_fill(pick->win, y, pick->x, pick->object[pidx],
526 pick->tbl_col_width);
527 }
528 }
529 }
530 pick_display_chyron(pick);
531}

550/** @brief Reverses the display of the currently selected object in pick window
551 @param pick Pointer to Pick structure containing object and display
552 information
553 @note Calculates the x coordinate for the currently selected object based on
554 the current table column and column width. Moves the cursor to the object's
555 position in the pick window, turns on reverse video attribute, and displays
556 the object's text. Turns off reverse video attribute and refreshes the pick
557 window to show the updated display. Moves the cursor back to the position
558 before the object text for potential further interactions. */
559void reverse_object(Pick *pick) {
560 pick->x = pick->tbl_col * (pick->tbl_col_width + 1) + 1;
561 pick->tbl_line = (pick->obj_idx / pick->tbl_cols) % pick->pg_lines;
562 pick->y = pick->tbl_line;
563 wmove(pick->win, pick->y, pick->x);
564 wattron(pick->win, WA_REVERSE);
565 mvwaddstr_fill(pick->win, pick->y, pick->x, pick->object[pick->obj_idx],
566 pick->tbl_col_width);
567 wattroff(pick->win, WA_REVERSE);
568 wrefresh(pick->win);
569 wmove(pick->win, pick->y, pick->x - 1);
570}

589/** @brief Toggles the selection state of the currently selected object in pick
590 window
591 @param pick Pointer to Pick structure containing object and selection
592 information
593 @note Calculates the x coordinate for the currently selected object based on
594 the current table column and column width. If the object is currently
595 selected, it is deselected by updating the selection count, marking it as not
596 selected, and displaying a space before the object text. If the object is not
597 currently selected, it is selected by updating the selection count, marking
598 it as selected, and displaying an asterisk before the object text. Refreshes
599 the pick window to show the updated display. Moves the cursor back to the
