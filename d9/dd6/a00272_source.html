339 break;
340 default:
341 break;
342 }
343 }
344 init->argv[0] = strdup(argv[0]);
345 i = 0;
346 while (i < argc) {
347 init->argv[i] = strdup(argv[i]);
348 i++;
349 }

351 init->argc = argc;

644int prompt_str_to_int(char *s) {
645 int prompt_type;
646 str_to_lower(s);
647 if (strcmp(s, "short") == 0)
648 prompt_type = PT_SHORT;
649 else if (strcmp(s, "long") == 0)
650 prompt_type = PT_LONG;
651 else if (strcmp(s, "string") == 0)
652 prompt_type = PT_STRING;
653 else
654 prompt_type = PT_NONE;
655 return prompt_type;
656}
